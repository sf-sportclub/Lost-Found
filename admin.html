/** * Cloudflare Worker v2: Secure API & Storage Gateway (Super Resilient Edition) * หน้าที่: ประสานงานเชื่อมต่อข้อมูลระหว่างหน้าเว็บหลัก, หน้าเว็บแอดมิน HR และฐานข้อมูล Supabase * คุณสมบัติเด่นของเวอร์ชันนี้: * 1. ป้องกันรหัสผ่านคลาดเคลื่อนด้วยการตรวจเช็กและตัดช่องว่าง (Auto-Trim) 🔑 * 2. รองรับด่านตรวจสิทธิ์แอดมิน 3 ช่องทางหลักเพื่อป้องกันเครือข่ายบล็อก Headers 🛡️ * 3. แจ้งเตือนข้อผิดพลาดจาก Supabase กลับไปที่หน้าเว็บอย่างชัดเจนเพื่อให้วิเคราะห์บั๊กได้ง่าย */ export default { async fetch(request, env) { const url = new URL(request.url); // 1. ตั้งค่า CORS Headers คุมความปลอดภัยระดับมาตรฐานสากล const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS, PATCH, DELETE", "Access-Control-Allow-Headers": "Content-Type, Authorization, X-Admin-Password, x-admin-password", }; // ตอบกลับ OPTIONS (Preflight) ของเบราว์เซอร์ทันทีเพื่อเปิดช่องการส่งข้อมูล if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } try { if (!env.SUPABASE_URL || !env.SUPABASE_KEY) { throw new Error("ระบบตรวจไม่พบตัวแปรระบบ SUPABASE_URL หรือ SUPABASE_KEY บน Cloudflare Worker Settings"); } // ดึงรหัสผ่านลับแอดมินจากระบบ พร้อมล้างเว้นวรรคส่วนเกินที่เกิดจากการคัดลอก (Auto-Trim) const ADMIN_PASSWORD = (env.ADMIN_PASSWORD || "sf-admin-1234").trim(); // ฟังก์ชันสแกนสิทธิ์แอดมินระบบอัจฉริยะ (สแกนตรวจสิทธิ์ 3 ช่องทางยืดหยุ่นสูงสุด) const authorizeAdmin = (req) => { // ช่องทางที่ 1: ตรวจสอบจากหัวจดหมาย X-Admin-Password (รองรับตัวพิมพ์เล็กและใหญ่) const rawHeader = req.headers.get("X-Admin-Password") || req.headers.get("x-admin-password"); if (rawHeader) { const clientPassword = rawHeader.split(',')[0].trim(); if (clientPassword === ADMIN_PASSWORD) return true; } // ช่องทางที่ 2: ตรวจสอบจากมาตรฐานสิทธิ์การเชื่อมต่อ Authorization Bearer const authHeader = req.headers.get("Authorization"); if (authHeader && authHeader.startsWith("Bearer ")) { const token = authHeader.substring(7).trim(); if (token === ADMIN_PASSWORD) return true; } // ช่องทางที่ 3: ตรวจสอบจากพารามิเตอร์ URL (Query Parameter) เผื่อระบบเครือข่ายบล็อก Headers const urlParamPass = url.searchParams.get("pass"); if (urlParamPass && urlParamPass.trim() === ADMIN_PASSWORD) { return true; } return false; }; // ========================================== // [หมวดหมู่ที่ 1: สำหรับผู้ใช้ทั่วไป (Public API)] // ========================================== // 1.1 ดึงรายการประกาศสถานะทั้งหมดเพื่อส่งกลับไปแสดงผล if (request.method === "GET" && url.pathname === "/api/posts") { const supabaseTarget = `${env.SUPABASE_URL}/rest/v1/posts?order=id.desc`; const response = await fetch(supabaseTarget, { method: "GET", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json" } }); if (!response.ok) { const errText = await response.text(); throw new Error(`Supabase Error: ${response.status} - ${errText}`); } const data = await response.json(); return new Response(JSON.stringify(data), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // 1.2 รับข้อมูลส่งฟอร์มประกาศใหม่ พร้อมอัปโหลดภาพเข้า Supabase Storage if (request.method === "POST" && url.pathname === "/api/posts") { const payload = await request.json(); const timestamp = new Date().toLocaleString("th-TH", { timeZone: "Asia/Bangkok" }); const imageUrls = []; if (payload.images && payload.images.length > 0) { for (let i = 0; i < payload.images.length; i++) { const img = payload.images[i]; const uniqueFilename = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}-${img.filename}`; const binaryString = atob(img.base64); const bytes = new Uint8Array(binaryString.length); for (let j = 0; j < binaryString.length; j++) { bytes[j] = binaryString.charCodeAt(j); } const uploadUrl = `${env.SUPABASE_URL}/storage/v1/object/sf-lostfound-images/${uniqueFilename}`; const uploadResponse = await fetch(uploadUrl, { method: "POST", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": img.mimeType, "x-upsert": "true" }, body: bytes.buffer }); if (uploadResponse.ok) { const publicUrl = `${env.SUPABASE_URL}/storage/v1/object/public/sf-lostfound-images/${uniqueFilename}`; imageUrls.push(publicUrl); } else { const errText = await uploadResponse.text(); throw new Error(`Failed to upload image to Supabase: ${errText}`); } } } const imagesString = imageUrls.join(','); const dbTarget = `${env.SUPABASE_URL}/rest/v1/posts`; const insertResponse = await fetch(dbTarget, { method: "POST", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json", "Prefer": "return=minimal" }, body: JSON.stringify({ timestamp: timestamp, type: payload.type, itemName: payload.itemName, category: payload.category, eventDate: payload.eventDate, location: payload.location, details: payload.details || "", contactName: payload.contactName, phone: payload.phone, lineId: payload.lineId || "-", images: imagesString, status: "ใช้งาน" }) }); if (!insertResponse.ok) { const errText = await insertResponse.text(); throw new Error(`Failed to insert record: ${errText}`); } return new Response(JSON.stringify({ status: "success", message: "บันทึกข้อมูลและรูปภาพสำเร็จ" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // ======================================= // [หมวดหมู่ที่ 2: สำหรับผู้ดูแลระบบ HR (Admin API)] // ======================================= // 2.1 ตรวจสอบคำร้องรหัสล็อกอินของฝ่ายบุคคล if (request.method === "POST" && url.pathname === "/api/admin/login") { const payload = await request.json(); const clientPassword = (payload.password || "").trim(); if (clientPassword === ADMIN_PASSWORD) { return new Response(JSON.stringify({ status: "success", message: "ตรวจสอบรหัสผ่านถูกต้อง" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } else { return new Response(JSON.stringify({ status: "error", message: "รหัสผ่านผู้ดูแลระบบไม่ถูกต้อง" }), { status: 401, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } // ด่านสแกนเช็กสิทธิ์ความปลอดภัยแอดมินก่อนเข้าถึง API จัดการตารางข้อมูลด้านล่าง if (url.pathname.startsWith("/api/admin/")) { if (!authorizeAdmin(request)) { return new Response(JSON.stringify({ status: "error", message: "สิทธิ์แอดมินไม่ถูกต้องหรือเซสชันหมดอายุ" }), { status: 403, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } // 2.2 ดึงข้อมูลทุกประกาศแบบไม่มีการคัดออกสำหรับแผงจัดการ if (request.method === "GET" && url.pathname === "/api/admin/posts") { const supabaseTarget = `${env.SUPABASE_URL}/rest/v1/posts?order=id.desc`; const response = await fetch(supabaseTarget, { headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json" } }); if (!response.ok) { const errText = await response.text(); throw new Error(`Supabase Error: ${response.status} - ${errText}`); } const data = await response.json(); return new Response(JSON.stringify(data), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // 2.3 แก้ไขสถานะประกาศ เช่น ปิดเคสเป็นสำเร็จ หรือกู้คืนโพสต์ if (request.method === "POST" && url.pathname === "/api/admin/posts/update-status") { const payload = await request.json(); const updateTarget = `${env.SUPABASE_URL}/rest/v1/posts?id=eq.${payload.id}`; const response = await fetch(updateTarget, { method: "PATCH", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ status: payload.status }) }); if (!response.ok) { const errText = await response.text(); throw new Error(`Supabase Update Error: ${errText}`); } return new Response(JSON.stringify({ status: "success", message: "ปรับสถานะสำเร็จ" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // 2.4 ลบข้อมูลโพสต์ออกจากตารางอย่างถาวร if (request.method === "POST" && url.pathname === "/api/admin/posts/delete") { const payload = await request.json(); const deleteTarget = `${env.SUPABASE_URL}/rest/v1/posts?id=eq.${payload.id}`; const response = await fetch(deleteTarget, { method: "DELETE", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json" } }); if (!response.ok) { const errText = await response.text(); throw new Error(`Supabase Delete Error: ${errText}`); } return new Response(JSON.stringify({ status: "success", message: "ลบโพสต์ถาวรสำเร็จ" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // 2.5 บันทึกข้อมูลที่แก้ไขตัวสะกดคำผิดของการ์ดประกาศ if (request.method === "POST" && url.pathname === "/api/admin/posts/edit") { const payload = await request.json(); const editTarget = `${env.SUPABASE_URL}/rest/v1/posts?id=eq.${payload.id}`; const response = await fetch(editTarget, { method: "PATCH", headers: { "apikey": env.SUPABASE_KEY, "Authorization": `Bearer ${env.SUPABASE_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ itemName: payload.itemName, category: payload.category, eventDate: payload.eventDate, location: payload.location, details: payload.details, contactName: payload.contactName, phone: payload.phone, lineId: payload.lineId }) }); if (!response.ok) { const errText = await response.text(); throw new Error(`Supabase Edit Error: ${errText}`); } return new Response(JSON.stringify({ status: "success", message: "แก้ไขรายละเอียดสำเร็จ" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } return new Response("SF Lost & Found Workers Gateway Running Successfully.", { status: 200, headers: { "Content-Type": "text/plain" } }); } catch (err) { return new Response(JSON.stringify({ status: "error", message: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }); } } };
